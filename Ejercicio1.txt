import java.util.*;
class Main {
  public static void main(String[] args) {

    Scanner teclado = new Scanner(System.in);

    int num1, num2;

    System.out.println("Introduce un numero");
    num1= teclado.nextInt();

    System.out.println("Introduce el numero por el cual quieres saber si es divisible");
    num2= teclado.nextInt();

    if (num1%num2 == 0){
       System.out.println("El numero " + num1 + " es divisible por el numero" + num2);
    }else {
      System.out.println("El numero " + num1 + " no es divisible por el numero" + num2);
    }
}
}
