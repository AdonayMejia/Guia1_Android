import java.util.*;

class Main {
  public static void main(String[] args) {
    Scanner teclado = new Scanner(System.in);

    int cuad1=0, cuad2=0, cuad3=0, cuad4=0, numcoor, i;

    System.out.println("Cuantas coordenadas desea ingresar");
    numcoor = teclado.nextInt();

    for (i = 1; i <= numcoor; i++) {
      System.out.println("Ingrese coordenada en x");
      double x = teclado.nextInt();
      System.out.println("Ingrese coordenada en y");
      double y = teclado.nextInt();

      if (x > 0 && y > 0) {
        cuad1++;
      } else {
        if (x < 0 && y > 0) {
          cuad2++;
        } else {
          if (x < 0 && y < 0) {
            cuad3++;
          } else {
            if (x > 0 && y < 0) {
              cuad4++;
            }
          }
        }
      }
    }
    System.out.println("\nCantidad de puntos ingresados en cuadrante 1: " + cuad1);
    System.out.println("Cantidad de puntos ingresados en cuadrante 2: " + cuad2);
    System.out.println("Cantidad de puntos ingresados en cuadrante 3: " + cuad3);
    System.out.println("Cantidad de puntos ingresados en cuadrante 4: " + cuad4);
  }
}
